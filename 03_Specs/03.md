  deg: number
  color: string

  const [rotation, setRotation] = useState(0)

  // –ü–æ–≤—ñ–ª—å–Ω–µ –æ–±–µ—Ä—Ç–∞–Ω–Ω—è –≤—Å—ñ—î—ó —Å–∏—Å—Ç–µ–º–∏

                  key={`${i}-${j}`}
                  x1={c1.x} y1={c1.y}
                  x2={c2.x} y2={c2.y}

        {/* –¶–µ–Ω—Ç—Ä (–ß–æ—Ä–Ω–∞ –¥—ñ—Ä–∞) */}
